

Naming Convensions:
------------------------
VARIBALE : 
	DECLARATION: V_VARIABLE_NAME
	EXAMPLE: V_EMPLOYEE_NAME
	
CURSOR:
	DECLARATION: CUR_CURSOR_NAME 
	EXAMPLE: CUR_EMPLOYEE_LIST
	
EXCEPTION: 
	DECLARATION: E_EXCEPTION_NAME
	EXAMPLE: E_INVALID_EMPLOYEE_NAME
	
PROCEDURE:
	DECLARATION: P_PROCEDURE_NAME
	EXAMPLE: P_EMPLOYEE_GET_EMPLOYEES_BY_FILTER
