

Statement & Row Level Triggers
---------------------------------

ROW LEVEL TRIGGER
------------------------
SET serveroutput ON;
/
CREATE OR REPLACE TRIGGER TRG_EMPLOYEES BEFORE
  INSERT OR
  UPDATE OR
  DELETE ON EMPLOYEE_COPY 
  FOR EACH ROW
  BEGIN 
  DBMS_OUTPUT.PUT_LINE('THIS IS TRIGGER ON ROW LEVEL - FOR EACH ROW');
  
END TRG_EMPLOYEES;


--NOW WE WILL UPDATE SOME VALUES
/

UPDATE EMPLOYEE_COPY SET SALARY = SALARY + 100
WHERE EMPLOYEE_ID=100;

/
DELETE FROM EMPLOYEE_COPY
WHERE EMPLOYEE_ID=100;

/
SELECT *FROM EMPLOYEE_COPY
WHERE EMPLOYEE_ID=100;
